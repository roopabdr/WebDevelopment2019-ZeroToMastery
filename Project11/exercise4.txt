//using prompt() and alert(), ask a user for their age.
// IF they say they are below 18, respond with:
// "Sorry, you are too young to drive this car. Powering off"

// IF they say they are 18, respond with:
// "Congratulations on your first year of driving. Enjoy the ride!"

// IF they say they are over 18, respond with:
// "Powering On. Enjoy the ride!"



var year = prompt("Enter your year of birth: ");
var age = 2018 - Number(year);

if (age < 18 && age > 0) {
	alert("Your age is "+age+". Sorry, you are too young to drive this car. Powering off");
} else if (age === 18) {
	alert("Your age is "+age+". Congratulations on your first year of driving. Enjoy the ride!");
} else if (age > 18) {
	alert("Your age is "+age+". Powering On. Enjoy the ride!");
} else {
	alert("Well, looks like we can't know your age! Sorry!");
}
